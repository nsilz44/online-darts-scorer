Comments0
http://127.0.0.1:8090/players/update

Adds a new player to player list

Headers

Content-Typeapplication/json

Body

raw (application/json)
{"Forename":"a","Surname":"a","Nickname":"a","ID":"2020050414536"}
Example RequestPOST players/update
var myHeaders = new Headers();
myHeaders.append("Content-Type", "application/json");

var raw = JSON.stringify({"Forename":"a","Surname":"a","Nickname":"a","ID":"2020050414536"});

var requestOptions = {
  method: 'POST',
  headers: myHeaders,
  body: raw,
  redirect: 'follow'
};

fetch("http://127.0.0.1:8090/players/update", requestOptions)
  .then(response => response.text())
  .then(result => console.log(result))
